SALOME_CONFIGURE_FILE(conf.py.in conf.py)

SET(_cmd_options -c ${CMAKE_CURRENT_BINARY_DIR} -b html -d doctrees ${CMAKE_CURRENT_SOURCE_DIR} docutils)

SALOME_GENERATE_ENVIRONMENT_SCRIPT(_cmd env_script "${SPHINX_EXECUTABLE}" "${_cmd_options}" CONTEXT blocFissure_DOC)

ADD_CUSTOM_TARGET(html_docs_blocFissure COMMAND ${_cmd})

INSTALL(CODE "EXECUTE_PROCESS(COMMAND \"${CMAKE_COMMAND}\" --build ${PROJECT_BINARY_DIR} --target html_docs_blocFissure)")
INSTALL(DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/docutils/
  DESTINATION ${SALOME_INSTALL_DOC}/gui/SMESH/blocFissure
  PATTERN ".buildinfo" EXCLUDE
)

SET(make_clean_files docutils doctrees)
SET_DIRECTORY_PROPERTIES(PROPERTIES ADDITIONAL_MAKE_CLEAN_FILES "${make_clean_files}")
